tomcat_dir=/usr/local/tomcat

stop_tomcat(){

}

   /bin/bash  $tomcat_dir/bin/startup.sh

deploy(){

   stop_tomcat

 echo "start_tomcat"  
   start_tomcat  
   sleep 10  
}  
 
rollback(){  
 echo "stop_tomcat"  
   stop_tomcat  
   sleep 2  
 
 echo "rollback"  
   rm -f $tomcat_dir/webapps/ROOT.war && rm -rf $tomcat_dir/webapps/ROOT

   start_tomcat  
   sleep 10

'deploy')

'rollback')  
   rollback

*)  
   echo "Usage: $0 {deploy | rollback}"
